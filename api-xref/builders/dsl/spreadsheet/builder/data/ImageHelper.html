19  
20  import builders.dsl.spreadsheet.builder.api.CellDefinition;
21  import builders.dsl.spreadsheet.builder.api.ImageCreator;
22  
23  import java.io.*;
24  import java.net.URI;
25  import java.net.URISyntaxException;
26  import java.util.Base64;
27  
28  class ImageHelper implements ImageCreator {
29  
30      private final CellNode cellNode;
31      private final String format;
32  
33      public ImageHelper(CellNode cellNode, String format) {
34          this.cellNode = cellNode;
35          this.format = format;
36      }
37  
38      @Override
39      public CellDefinition from(String fileOrUrl) {
40          if (fileOrUrl.startsWith("file://")) {
41              try (FileInputStream fis = new FileInputStream(new File(new URI(fileOrUrl)))) {
42                  return from(fis);
43              } catch (URISyntaxException | IOException e) {
44                  throw new RuntimeException("Exception reading file: " + fileOrUrl, e);
45              }
46          }
47          MapNodeilder/data/MapNode.html#MapNode">MapNode mapNode = new MapNode();
48          mapNode.set("type", format);
49          mapNode.set("url", fileOrUrl);
50          cellNode.node.set("image", mapNode);
51          return cellNode;
52      }
53  
54      @Override
55      public CellDefinition from(InputStream stream) {
56          return from(streamToBytes(stream));
57      }
58  
59      @Override
60      public CellDefinition from(byte[] imageData) {
61          return from("data:image/" + format + ";base64," + Base64.getEncoder().encodeToString(imageData));
62      }
63  
64      private byte[] streamToBytes(InputStream is) {
65          try {
66              ByteArrayOutputStream buffer = new ByteArrayOutputStream();
67              int nRead;
68              byte[] data = new byte[1024];
69              while ((nRead = is.read(data, 0, data.length)) != -1) {
70                  buffer.write(data, 0, nRead);
71              }
72  
73              buffer.flush();
74              return buffer.toByteArray();
75          } catch (IOException e) {
76              throw new RuntimeException("Exception reading image stream: " + is, e);
77          }
78      }
79  }
