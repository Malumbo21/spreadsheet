18  package builders.dsl.spreadsheet.impl;
19  
20  import builders.dsl.spreadsheet.builder.api.*;
21  
22  import java.util.ArrayList;
23  import java.util.LinkedHashMap;
24  import java.util.List;
25  import java.util.Map;
26  import java.util.function.Consumer;
27  
28  public abstract class AbstractWorkbookDefinition implements WorkbookDefinition {
29  
30      private final Map<String, Consumer<CellStyleDefinition>> namedStylesDefinition = new LinkedHashMap<String, Consumer<CellStyleDefinition>>();
31      private final Map<String, AbstractCellStyleDefinition> namedStyles = new LinkedHashMap<String, AbstractCellStyleDefinition>();
32      private final Map<String, AbstractSheetDefinition> sheets = new LinkedHashMap<String, AbstractSheetDefinition>();
33      private final List<Resolvable> toBeResolved = new ArrayList<Resolvable>();
34  
35      @Override
36      public final WorkbookDefinition style(String name, Consumer<CellStyleDefinition> styleDefinition) {
37          namedStylesDefinition.put(name, styleDefinition);
38          return this;
39      }
40  
41      // TODO: make package private again
42      public final void resolve() {
43          for (Resolvable resolvable : toBeResolved) {
44              resolvable.resolve();
45          }
46      }
47  
48      protected abstract AbstractCellStyleDefinition createCellStyle();
49      protected abstract AbstractSheetDefinition createSheet(String name);
50  
51      @Override
52      public final WorkbookDefinition sheet(String name, Consumer<SheetDefinition> sheetDefinition) {
53          AbstractSheetDefinition sheet = sheets.get(name);
54  
55          if (sheet == null) {
56              sheet = createSheet(name);
57              sheets.put(name, sheet);
58          }
59  
60          sheetDefinition.accept(sheet);
61  
62          sheet.resolve();
63  
64          return this;
65      }
66  
67      final AbstractCellStyleDefinition getStyles(Iterable<String> names) {
68          String name = Utils.join(names, ".");
69  
70          AbstractCellStyleDefinition style = namedStyles.get(name);
71  
72          if (style != null) {
73              return style;
74          }
75  
76          style = createCellStyle();
77          for (String n : names) {
78              getStyleDefinition(n).accept(style);
79          }
80  
81          style.seal();
82  
83          namedStyles.put(name, style);
84  
85          return style;
86      }
87  
88      final Consumer<CellStyleDefinition> getStyleDefinition(String name) {
89          Consumer<CellStyleDefinition> style = namedStylesDefinition.get(name);
90          if (style == null) {
91              throw new IllegalArgumentException("Style '" + name + "' is not defined");
92          }
93          return style;
94      }
95  
96      void addPendingFormula(AbstractPendingFormula formula) {
97          toBeResolved.add(formula);
98      }
99  
100     protected void addPendingLink(AbstractPendingLink link) {
101         toBeResolved.add(link);
102     }
103 }
